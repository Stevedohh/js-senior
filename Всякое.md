```js

export const useLogger = () => { 

	const isFirst = useRef(true) 
	
	useEffect(() => { 
		if(!isFirst.current)
			console.log('update') 
	}) 

	useEffect(() => { 
		console.log('mount') 
		isFirst.current = false; 
		return () => console.log('unmount') 
	}, [])
} 
```
